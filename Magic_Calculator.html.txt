    <script>
        function calculate() {
            // Get values from inputs
            const yourAmount = parseFloat(document.getElementById("yourAmount").value) || 0;
            const beggarAmount = parseFloat(document.getElementById("beggarAmount").value) || 0;

            // Calculate values
            const mine = yourAmount;
            const sum = yourAmount + mine + beggarAmount;
            const halfToBeggar = sum / 2;
            const giveMeMine = mine;
            const remaining = sum - halfToBeggar - giveMeMine;

            // Display results
            document.getElementById("mine").textContent = mine;
            document.getElementById("sum").textContent = sum;
            document.getElementById("halfToBeggar").textContent = halfToBeggar;
            document.getElementById("giveMeMine").textContent = giveMeMine;
            document.getElementById("remaining").textContent = remaining;
        }
    </script>
